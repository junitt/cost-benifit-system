$.ajax({
    type: "POST",
    url: "/login",
    data: {
        username: "admin",
        password: "password"
    },
    success: function (response) {
        if (response === "success") {
            // 登录成功，跳转到其它页面
            window.location.href = "/home";
        } else {
            // 登录失败，提示错误
            alert("用户名或密码错误");
        }
    }
});
